### Numbers
```cpp
cout << 0 << endl; // 0
cout << 1 << endl; // 1
cout << -1 << endl; // -1
cout << 0.5 << endl; // 0.5
cout << .1 << endl; // 0.1
cout << 1. << endl; // 1.0
cout << 1e3 << endl; // 1000
cout << 1.23e-3 << endl; // 0.00123
cout << 1 + 1 << endl; // 2
cout << 4 - 7 << endl; // -3
cout << 3 * 4 << endl; // 12
cout << 1.2 * 5.6 << endl; // 6.72 
cout << 3 * 1.2 << endl; // 3.6
cout << 6 / 2 << endl; // 3
cout << 4 / 3 << endl; // 1
cout << 4.0 / 3.0 << endl; // 1.33333
cout << 4 / 3.0 << endl; // 1.33333
cout << 4.0 / 3 << endl; // 1.33333
cout << 12 / 0 << endl; // Error: Floating point exception (core dumped)
cout << 11 % 5 << endl; // 1
cout << 12 % 6 << endl; // 0
cout << 12 % 0 << endl; // Error: Floating point exception (core dumped)
```
