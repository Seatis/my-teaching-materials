### Functions
```python
def greet_fox():
  print("Hello Green Fox!")

greet_fox()
greet_fox()


# function arguments

def greet_by_name(name):
  print("Szia,", name)

greet_by_name("Tojas")
greet_by_name("Barbi")


# default values for function arguments

def greet(greet="Szia", name="Valaki"):
  print(greet, name)

greet("hello", "Tojas")
greet("szevasz", "Barbi")
greet("csa")
greet(name="Mindenki")


# function return values

def make_green(name):
  new_name = "Zold " + name
  return new_name

name = make_green("Tojas")
greet_by_name(name)

```

### Function scope
```python
# global scope
n = 2

def f():
  # local scope
  n = 1
  print("local", n)

f() # => 1
print("global", n)

def c():
  r = 1
  print(r)
c()
print(r)

```

### Data structures
```python
giraffe = {'color': 'yellow', 'pattern': 'polygonal patches'}
whale = {'color': 'blue', 'size': 10000}

print(giraffe)
print(giraffe['pattern'])
print(whale['size'])

whale['say'] = 'eeeeeeeeeeeeuuuuuuw'

print(whale)
```
